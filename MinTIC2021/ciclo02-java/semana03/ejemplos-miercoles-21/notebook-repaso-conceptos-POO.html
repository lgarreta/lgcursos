Ejemplo de Animales¶

In [5]:
public abstract class Animal {
    public abstract void hacerRuido ();
}

public class Perro extends Animal {
    public void hacerRuido () {
        System.out.println ("guau guau");
    }
    
    public void hacerRuido (String estado) {
        if (estado == "contento")
            System.out.println ("Mueve la cola y dice guau guau");
        else if (estado == "bravo")
            System.out.println ("Muerde y dice GUAU GUAU");
        else
            hacerRuido ();      
    }
}

public class Gato extends Animal {
    public void hacerRuido () {
        System.out.println ("miau miau");
    }    
}

public class Vaca extends Animal {
    public void hacerRuido () {
        System.out.println ("muu muu");
    }    
}
public class Tigre extends Animal {
    public void hacerRuido () {
        System.out.println ("grurrr grurrr");
    }    
}
